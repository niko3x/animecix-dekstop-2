    **2. Update `animecix-v2/src/main.ts`:**

    The import of `tau-protocol.ts` MUST be at the top of the file because it has a module-level side effect (`protocol.registerSchemesAsPrivileged`) that must run before `app.whenReady()`.

    ```typescript
    import { app, BrowserWindow, ipcMain, session, net } from 'electron';
    import started from 'electron-squirrel-startup';
    // CRITICAL: This import triggers protocol.registerSchemesAsPrivileged at module level
    import './player/tau-protocol';
    import { registerTauProtocol } from './player/tau-protocol';
    import { StorageService } from './storage/StorageService';
    import { createWindow } from './window/WindowService';
    import { registerWindowIpc } from './window/window.ipc';
    import { AdBlocker } from './network/ad-blocker';
    import { setupRequestInterception } from './network/request-handler';
    import { setupHeaderRewriter } from './network/header-rewriter';
    import { registerDeepLinkProtocol, extractDeepLinkFromArgs, handleDeepLink } from './auth/deep-link';
    import { DiscordService } from './integrations/discord-rpc';

    if (started) { app.quit(); }

    let mainWindow: BrowserWindow | null = null;
    let storage: StorageService | null = null;
    let discord: DiscordService | null = null;

    // Track current episode for Discord RPC -- populated by API response interception
    let currentEpisode: { title: string; seasonNumber?: string; episodeNumber?: string; translator?: string; posterUrl?: string; animeId?: string } | null = null;

    // Buffer deep link URL for cold start (process before window is ready)
    let pendingDeepLink: string | null = extractDeepLinkFromArgs(process.argv);

    const gotLock = app.requestSingleInstanceLock();

    if (!gotLock) {
      app.quit();
    } else {
      // Register deep link protocol
      registerDeepLinkProtocol();

      // macOS deep link handler (warm start)
      app.on('open-url', (event, url) => {
        event.preventDefault();
        if (mainWindow) {
          handleDeepLink(url, mainWindow.webContents);
          if (mainWindow.isMinimized()) mainWindow.restore();
          mainWindow.focus();
        } else {
          pendingDeepLink = url;
        }
      });

      // Windows deep link + second instance (warm start)
      app.on('second-instance', (_event, commandLine) => {
        if (mainWindow) {
          // Check for deep link in command line
          const deepLinkUrl = extractDeepLinkFromArgs(commandLine);
          if (deepLinkUrl) {
            handleDeepLink(deepLinkUrl, mainWindow.webContents);
          }
          if (mainWindow.isMinimized()) mainWindow.restore();
          mainWindow.focus();
        }
      });

      app.whenReady().then(() => {
        storage = new StorageService();

        // Register tau-player:// protocol handler
        registerTauProtocol();

        // Set up network interception (combined iframe redirect + ad blocker)
        const adBlocker = new AdBlocker();
        setupRequestInterception(adBlocker);
        setupHeaderRewriter();

        // --- Episode metadata interception ---
        // Intercept tau-video.xyz API responses to extract episode metadata for Discord RPC.
        // When a request to tau-video.xyz/api/video/* completes, fetch the same URL to read
        // the Video JSON (title_id, season_number, episode_number, translator, poster).
        // This resolves Open Question #5 from RESEARCH.md.
        session.defaultSession.webRequest.onCompleted(
          { urls: ['*://tau-video.xyz/api/video/*'] },
          (details) => {
            if (details.statusCode !== 200) return;
            // Fetch the same URL from main process to read the response body
            // (webRequest.onCompleted does not provide response body directly)
            net.fetch(details.url, {
              method: 'GET',
              // Uses the same session cookies automatically via default session
            }).then(async (response) => {
              if (!response.ok) return;
              const video = await response.json() as {
                title_id?: string;
                season_number?: string;
                episode_number?: string;
                translator?: string;
              };
              // Get human-readable title from the main window's page title
              const pageTitle = mainWindow?.webContents.getTitle() ?? '';
              // animecix.tv page titles typically contain the anime name
              // e.g., "Naruto Shippuuden 1. Sezon 5. Bolum izle - AnimeciX"
              // Extract the part before " izle" or use the full title
              const animeTitle = pageTitle.split(' izle')[0]?.trim() || pageTitle.split(' - ')[0]?.trim() || 'Anime';

              currentEpisode = {
                title: animeTitle,
                seasonNumber: video.season_number,
                episodeNumber: video.episode_number,
                translator: video.translator,
                animeId: video.title_id,
              };

              // Update Discord RPC with new episode
              if (discord) {
                discord.updateActivity({
                  ...currentEpisode,
                  isPlaying: true,
                  startTimestamp: Math.floor(Date.now() / 1000),
                });
              }

              // Send the anime ID to preload so captionsChanged can include it
              if (mainWindow && video.title_id) {
                mainWindow.webContents.send('player:setAnimeId', video.title_id);
              }
            }).catch(() => {
              // Silent fail -- API fetch for metadata is best-effort
            });
          }
        );

        // Create window
        mainWindow = createWindow(storage);
        registerWindowIpc(mainWindow);

        // Initialize Discord RPC
        discord = new DiscordService();

        // IPC: player state updates from preload (for Discord RPC)
        ipcMain.on('player:state', (_event, data: { time: number; duration: number; isPlaying: boolean }) => {
          if (discord && currentEpisode) {
            discord.updateActivity({
              ...currentEpisode,
              isPlaying: data.isPlaying,
              startTimestamp: data.isPlaying ? Math.floor(Date.now() / 1000) - Math.floor(data.time) : undefined,
            });
          }
        });

        // IPC: subtitle change from preload (with tracked animeId)
        ipcMain.on('player:captionsChanged', (_event, data: { animeId: string; language: string }) => {
          if (storage && data.animeId) {
            storage.setSubtitlePref(data.animeId, data.language);
          }
        });

        // IPC: get subtitle preference for anime
        ipcMain.handle('player:getSubtitlePref', (_event, animeId: string) => {
          return storage?.getSubtitlePref(animeId) ?? 'tr';
        });

        // Process pending deep link (cold start)
        if (pendingDeepLink && mainWindow) {
          const win = mainWindow;
          win.webContents.once('did-finish-load', () => {
            handleDeepLink(pendingDeepLink!, win.webContents);
            pendingDeepLink = null;
          });
        }
      }).catch((err) => {
        console.error('Failed to initialize app:', err);
        app.quit();
      });

      app.on('window-all-closed', () => {
        if (process.platform !== 'darwin') {
          discord?.destroy();
          discord = null;
          storage?.close();
          storage = null;
          app.quit();
        }
      });

      app.on('activate', () => {
        if (BrowserWindow.getAllWindows().length === 0 && storage) {
          mainWindow = createWindow(storage);
          registerWindowIpc(mainWindow);
        }
      });

      app.on('before-quit', () => {
        discord?.destroy();
        discord = null;
        storage?.close();
        storage = null;
      });
    }
    ```

    Add a window 'message' event listener that observes postMessage events from the tau iframe and forwards relevant ones to main:

    ```typescript
    // PostMessage observer -- forwards player state to main for Discord RPC + subtitle prefs
    window.addEventListener('message', (event) => {
      // Only accept messages from tau-player:// or tau-video.xyz origins
      if (!event.origin.startsWith('tau-player://') && !event.origin.includes('tau-video.xyz')) return;

      const data = event.data;
      if (!data || typeof data.action !== 'string') return;

      switch (data.action) {
        case 'currentTime':
          ipcRenderer.send('player:state', {
            time: data.time,
            duration: data.duration,
            isPlaying: data.isPlaying,
          });
          break;
        case 'play':
          ipcRenderer.send('player:state', { time: 0, duration: 0, isPlaying: true });
          break;
        case 'pause':
          ipcRenderer.send('player:state', { time: 0, duration: 0, isPlaying: false });
          break;
        case 'captionsChanged':
          // Use tracked currentAnimeId from main process (set via player:setAnimeId IPC)
          // so subtitle preferences are stored per-anime, not with empty key
          ipcRenderer.send('player:captionsChanged', {
            animeId: currentAnimeId,
            language: String(data.track),
          });
          break;
      }
    });
    ```
