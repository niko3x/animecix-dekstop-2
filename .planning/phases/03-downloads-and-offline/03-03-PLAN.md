    1. Create `animecix-v2/src/cache/CacheEvictor.ts`:

    ```typescript
    import { StorageService } from '../storage/StorageService';
    import fs from 'node:fs';

    export class CacheEvictor {
      private storage: StorageService;
      private maxBytes: number;

      constructor(storage: StorageService, defaultMaxBytes: number = 10 * 1024 * 1024 * 1024) {
        this.storage = storage;
        // Read user-configured cap from settings, fall back to default (10 GB per D-06)
        const saved = storage.getSetting('cache_max_bytes');
        this.maxBytes = saved ? parseInt(saved, 10) : defaultMaxBytes;
      }

      setMaxBytes(maxBytes: number): void {
        this.maxBytes = maxBytes;
        this.storage.setSetting('cache_max_bytes', String(maxBytes));
      }

      getMaxBytes(): number {
        return this.maxBytes;
      }

      /**
       * Evicts oldest cache entries until total cache size is under maxBytes.
       * Deletes MP4 and subtitle files from disk for each evicted entry.
       * Returns array of evicted episodeIds.
       */
      evictIfNeeded(): string[] {
        const evictedEntries = this.storage.evictOldestCache(this.maxBytes);
        // evictOldestCache returns { episodeId, mp4Path, subPaths }[] (updated in Plan 01)
        // For each evicted entry, delete disk files:
        for (const entry of evictedEntries) {
          const subPaths = JSON.parse(entry.subPaths || '[]') as { language: string; path: string }[];
          this.deleteFilesForEntry(entry.mp4Path, subPaths);
        }
        return evictedEntries.map((e) => e.episodeId);
      }

      /**
       * Delete files for a specific cache entry from disk.
       */
      deleteFilesForEntry(mp4Path: string, subPaths: { language: string; path: string }[]): void {
        try { fs.unlinkSync(mp4Path); } catch { /* file may not exist */ }
        for (const sub of subPaths) {
          try { fs.unlinkSync(sub.path); } catch { /* file may not exist */ }
        }
      }
    }
    ```

    **NOTE:** Plan 01 defines `evictOldestCache` to return `{ episodeId: string; mp4Path: string; subPaths: string }[]` (not `string[]`). This provides the file paths needed for disk cleanup. CacheEvictor.evictIfNeeded() iterates the returned entries and calls `this.deleteFilesForEntry(entry.mp4Path, JSON.parse(entry.subPaths))` for each.

    import { StorageService } from '../storage/StorageService';
    import { CacheEvictor } from './CacheEvictor';
    import { HlsMuxer } from './HlsMuxer';
    import https from 'node:https';

    import fs from 'node:fs';

    export class StreamCache extends EventEmitter {
      private storage: StorageService;
      private evictor: CacheEvictor;
      private cacheDir: string;
      private isCaching: boolean = false;

      // --- Transparent auto-caching state (D-05) ---
      // Tracks video segment URLs per episode as they complete via session.webRequest.onCompleted
      private currentEpisodeId: string | null = null;
      private segmentUrls: Map<string, string[]> = new Map(); // episodeId -> segment URLs

      constructor(storage: StorageService, cacheDir: string) {
        super();
        this.storage = storage;
        this.cacheDir = cacheDir;
        this.evictor = new CacheEvictor(storage);
        fs.mkdirSync(cacheDir, { recursive: true });
      }

    **TWO CACHING PATHS (per D-05 -- "Do both"):**

    **Path 1 -- TRANSPARENT auto-caching via session.webRequest.onCompleted (D-05 locked, NO player changes):**

    Method `setupTransparentCaching(session: Electron.Session): void`:
    - Register `session.webRequest.onCompleted` listener that detects completed video segment requests
    - Filter pattern: URLs matching `*.ts` (HLS segments) or large MP4 byte-range responses from known video CDN domains (e.g., `*tau-video.xyz*`, `*.m3u8*`)
    - On each completed video request:
      - Call `this.onSegmentCompleted(url, details.statusCode)` to track the URL
      - Only track successful requests (statusCode 200 or 206)

    Method `setCurrentEpisode(episodeId: string, subs: { language: string; url: string }[]): void`:
    - Called from main.ts when website sends `episode:watched` or video data is fetched
    - Sets `this.currentEpisodeId = episodeId` and stores subs for later use
    - Clears previous segment tracking for this episode

    Method `onSegmentCompleted(url: string, statusCode: number): void`:
    - If no currentEpisodeId is set, ignore
    - If statusCode is not 200/206, ignore
    - If URL matches video segment pattern (`.ts` extension, or video MIME from CDN):
      - Append URL to `this.segmentUrls.get(currentEpisodeId)` list
      - Track in memory only (lightweight, no disk writes during playback)

    Method `finalizeEpisodeCache(episodeId: string): void`:
    - Called when episode playback ends (detected via URL pattern change / new episode load / explicit IPC)
    - If `this.segmentUrls.has(episodeId)` and segment count > 0:
      - Skip if already cached or downloaded
      - Trigger background re-fetch: download all tracked segment URLs to `cacheDir/episodeId/` temp dir
      - After all segments downloaded, call HlsMuxer to produce single MP4
      - Download subtitles alongside
      - Call `storage.addCacheEntry(...)` with final paths and size
      - Emit 'cached' event
    - This is fully transparent -- the player never knew about any of this

    **Path 2 -- Explicit `cache:episode` IPC (user-initiated):**

    Method `cacheEpisode(episodeId: string, videoUrl: string, isHls: boolean, subs: { language: string; url: string }[]): Promise<void>`:
    - Skip if already cached (check `storage.getCacheEntry(episodeId)`)
    - Skip if already downloaded (check `storage.getDownloadById(episodeId)` with status 'completed')
    - Call `evictor.evictIfNeeded()` to make space
    - Create episode cache directory: `path.join(cacheDir, episodeId)`
    - If `isHls`:
      - Download HLS playlist, parse segment URLs, download all .ts segments to temp dir
      - Call `HlsMuxer.mux(segmentPaths, outputMp4Path)` to produce single MP4
      - Delete temp .ts segment files
    - If not HLS (direct MP4 URL):
      - Download MP4 to `path.join(cacheDir, episodeId, 'video.mp4')` using native https.get with pipe to createWriteStream
    - Download all subtitle files: for each sub, download to `path.join(cacheDir, episodeId, sub.language + '.ass')`
    - Compute total size via `fs.statSync` on all files
    - Call `storage.addCacheEntry({ episodeId, mp4Path, subPaths: JSON.stringify(subPathsArray), sizeBytes })`
    - Emit 'cached' event with episodeId

    **Key design for transparent caching:**
    - The `session.webRequest.onCompleted` listener is PASSIVE -- it only observes completed requests, never modifies them
    - No player changes needed -- the player streams normally, StreamCache just watches what URLs were fetched
    - Background re-fetch happens AFTER playback, not during (no bandwidth competition)
    - If a segment re-fetch fails, the episode is simply not cached (graceful degradation)

    ```typescript
    import { execFile } from 'node:child_process';
    import { promisify } from 'node:util';
    import fs from 'node:fs';
    import path from 'node:path';

    const execFileAsync = promisify(execFile);

    /**
     * Finds the ffmpeg binary path.
     * In dev: uses system ffmpeg (expected at /opt/homebrew/bin/ffmpeg or in PATH).
     * In packaged app: will use ffmpeg-static or extraResources (Phase 4 concern).
     */
    export function findFfmpegPath(): string {
      // Try ffmpeg-static first (if installed)
      try {
        return require('ffmpeg-static') as string;
      } catch {
        // Fall back to system ffmpeg
        return 'ffmpeg';
      }
    }

    /**
     * Check if ffmpeg is available on the system.
     */
    export async function isFfmpegAvailable(): Promise<boolean> {
      try {
        const ffmpegPath = findFfmpegPath();
        await execFileAsync(ffmpegPath, ['-version']);
        return true;
      } catch {
        return false;
      }
    }

    /**
     * Parse an HLS m3u8 playlist and extract segment URLs.
     * Handles both absolute and relative URLs.
     */
    export function parseM3u8(playlistContent: string, playlistUrl: string): string[] {
      const baseUrl = playlistUrl.substring(0, playlistUrl.lastIndexOf('/') + 1);
      const lines = playlistContent.split('\n').map(l => l.trim()).filter(l => l && !l.startsWith('#'));
      return lines.map(line => {
        if (line.startsWith('http://') || line.startsWith('https://')) {
          return line;
        }
        return baseUrl + line;
      });
    }

    /**
     * Mux TS segment files into a single MP4 using ffmpeg concat demuxer.
     * No re-encoding -- fast copy mode with faststart for streaming.
     *
     * SECURITY: Uses execFile (not exec) to prevent shell injection.
     * segmentPaths must be validated absolute paths.
     */
    export async function muxTsToMp4(segmentPaths: string[], outputMp4Path: string): Promise<void> {
      const ffmpegPath = findFfmpegPath();

      // Create concat file list for ffmpeg
      const concatDir = path.dirname(outputMp4Path);
      const concatListPath = path.join(concatDir, '_concat_list.txt');
      const concatContent = segmentPaths
        .map(p => `file '${p.replace(/'/g, "'\\''")}'`)
        .join('\n');
      fs.writeFileSync(concatListPath, concatContent);

      try {
        await execFileAsync(ffmpegPath, [
          '-f', 'concat',
          '-safe', '0',
          '-i', concatListPath,
          '-c', 'copy',
          '-movflags', 'faststart',
          '-y',  // overwrite output
          outputMp4Path,
        ], { timeout: 300000 }); // 5 min timeout for large files
      } finally {
        // Clean up concat list file
        try { fs.unlinkSync(concatListPath); } catch { /* ignore */ }
      }
    }

    /**
     * Manual TS concatenation fallback (no ffmpeg required).
     * Simple byte-level concatenation of MPEG-TS files.
     * Less reliable than ffmpeg for malformed segments but works for simple cases.
     */
    export async function concatTsToMp4Fallback(segmentPaths: string[], outputPath: string): Promise<void> {
      const output = fs.createWriteStream(outputPath);
      for (const segPath of segmentPaths) {
        await new Promise<void>((resolve, reject) => {
          const input = fs.createReadStream(segPath);
          input.pipe(output, { end: false });
          input.on('end', resolve);
          input.on('error', reject);
        });
      }
      output.end();
      await new Promise<void>((resolve) => output.on('finish', resolve));
    }
    ```

    Key design decisions:
    - Use `execFile` (NOT `exec`) to prevent shell injection (T-03 threat model)
    - Provide `concatTsToMp4Fallback` for environments without ffmpeg
    - `parseM3u8` handles relative and absolute segment URLs
    - 5-minute timeout on ffmpeg to prevent hanging on large files
    - `findFfmpegPath` tries ffmpeg-static first, falls back to system ffmpeg
